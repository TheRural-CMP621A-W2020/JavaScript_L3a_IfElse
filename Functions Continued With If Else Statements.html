Suppose that you want to randomly generate a number between 1 and 3. If the computer picks a 2 a window alert box will appear and say you won. Otherwise it will state that you lost and what the number was that it picked.


function CheckGuessFunction(){
var guess = (Math.floor(Math.random() * 3) + 1); //creates a ransom number between 1 and 3

if (guess==2) {
window.alert("Congratulations, you won!!");
}
else {
window.alert("Sorry you lost, the computer guessed a "+guess);
}
}
